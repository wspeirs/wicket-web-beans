1085     {

1087         int idx = identifier.lastIndexOf('.');
1088         if (idx < 0) {
1089             idx = identifier.lastIndexOf('$'); // Java nested classes.
1090         }
1091 
1092         if (idx >= 0 && identifier.length() > 1) {
1093             identifier = identifier.substring(idx + 1);
1094         }
1095 
1096         if (identifier.length() == 0) {
1097             return "";
1098         }

1100         char[] chars = identifier.toCharArray();
1101         StringBuffer buf = new StringBuffer(chars.length + 10);
1102 
1103         // Capitalize the first letter.
1104         buf.append(Character.toUpperCase(chars[0]));
1105         boolean lastLower = false;
1106         for (int i = 1; i < chars.length; ++i) {
1107             if (!Character.isLowerCase(chars[i])) {
1108                 // Lower to upper case transition -- add space before it
1109                 if (lastLower) {
1110                     buf.append(' ');
1111                 }
1112             }
1113 
1114             buf.append(chars[i]);
1115             lastLower = Character.isLowerCase(chars[i]) || Character.isDigit(chars[i]);
1116         }
1117 
1118         return buf.toString();
1119     }
