052 /**
053  * Registers field and viewer components. 
054  * The components must implement the following constructor, at minimum:<p>
055  *    XyzComponent(String wicketId, IModel model, PropertyMetaData metaData) 
056  *    
057  * When registering an array type, use Object[].class as the type, and specify an element type.
058  * 
059  * @author Dan Syrstad
060  */
061 public class ComponentRegistry implements Serializable
062 {
063     private Class[] constructorArgs = new Class[] { String.class, IModel.class, ElementMetaData.class, Boolean.TYPE };
064     
065     // Key is Target Type's class name (e.g., java.util.Date). Value is the
066     // Wicket Component (Field) class name. If the mapping contains an element type, the key has a suffix of
067     // '[' followed by the element type name.
068     private HashMap<String,String> registry;
069 
070     /**
071      * Construct a ComponentRegistry with the default component mappings. 
072      *
073      */
074     public ComponentRegistry()
075     {
076         registry = new HashMap<String, String>();
077 
078         register(Object.class, BeanGridField.class);
079         register(String.class, InputField.class);
080         register(Boolean.class, BooleanField.class);
081         register(Boolean.TYPE, BooleanField.class);
082         register(Character.class, InputField.class);
083         register(Character.TYPE, InputField.class);
084         register(Byte.class, InputField.class);
085         register(Byte.TYPE, InputField.class);
086         register(Short.class, InputField.class);
087         register(Short.TYPE, InputField.class);
088         register(Integer.class, InputField.class);
089         register(Integer.TYPE, InputField.class);
090         register(Long.class, InputField.class);
091         register(Long.TYPE, InputField.class);
092         register(Float.class, InputField.class);
093         register(Float.TYPE, InputField.class);
094         register(Double.class, InputField.class);
095         register(Double.TYPE, InputField.class);
096         register(BigInteger.class, InputField.class);
097         register(BigDecimal.class, InputField.class);
098         register(Date.class, DateTimeField.class);
099         register(java.sql.Date.class, DateTimeField.class);
100         register(Time.class, DateTimeField.class);
101         register(Timestamp.class, DateTimeField.class);
102         register(Calendar.class, DateTimeField.class);
103         register(Enum.class, JavaEnumField.class);
104         register(Object[].class, Enum.class, MultiSelectEnumField.class);
105         register(List.class, Enum.class, MultiSelectEnumField.class);
106

108         
109         // Register the following so that they're available for findMatchingFieldClass(), but not really available otherwise.
110         register(BeanGridField.class, BeanGridField.class);
111         register(BeanInCollapsibleField.class, BeanInCollapsibleField.class);
112         register(BeanInlineField.class, BeanInlineField.class);
113         register(BeanTableField.class, BeanTableField.class);
114         register(BeanWithParentLabelField.class, BeanWithParentLabelField.class);
115         register(EmptyField.class, EmptyField.class);
116         register(TextAreaField.class, TextAreaField.class);
117     }

129     /**
130      * Registers an Field in a type-safe fashion.
131      *
132      * @param targetType
133      * @param fieldComponent
134      */
135     public void register(Class targetType, Class<? extends Field> fieldComponent)
136     {
137         register(targetType.getName(), null, fieldComponent.getName());
138     }
139 
140     /**
141      * Registers an Field with an element type in a type-safe fashion.
142      *
143      * @param targetType
144      * @param fieldComponent
145      */
146     public void register(Class targetType, Class elementType, Class<? extends Field> fieldComponent)
147     {
148         register(targetType.getName(), elementType.getName(), fieldComponent.getName());
149     }

151     /**
152      * Registers an field in a non-type-safe fashion.
153      *
154      * @param targetTypeClassName
155      * @param elemenTypeName the element type name. May be null.
156      * @param fieldComponentClassName
157      */
158     public void register(String targetTypeClassName, String elemenTypeName, String fieldComponentClassName)
159     {
160         if (elemenTypeName != null) {
161             targetTypeClassName += '[' + elemenTypeName;
162         }
163         
164         registry.put(targetTypeClassName, fieldComponentClassName);
165     }
166     
167     /**
168      * Given a shortened version of a Field class name (e.g., TextAreaField), try to find a matching full class name in the
169      * registry. The first one found is returned.
170      *
171      * @param shortName the short class name.
172      * 
173      * @return the full class name, or null if not found.
174      */
175     public String findMatchingFieldClass(String shortName)
176     {
177         String matchStr = '.' + shortName;
178         for (String fieldClassName : registry.values()) {
179             if (fieldClassName.endsWith(matchStr)) {
180                 return fieldClassName;
181             }
182         }
183         
184         return null;
185     }
186     
187     /**
188      * Attempts to find the component class name for a given type and elementType.
189      * 
190      * @param type
191      * @param elementType the element type, which may be null.
192      * 
193      * @return the class name, or null if not found.
194      */
195     private String getComponentClassName(Class type, Class elementType)
196     {
197         String baseKey = type.getName();
198         
199         for (; elementType != null; elementType = elementType.getSuperclass()) {
200             String elementBaseKey = baseKey + '[';
201             // Search up class hierarchy for matching type.
202             String componentClassName = registry.get(elementBaseKey + elementType.getName());
203             if (componentClassName != null) {
204                 return componentClassName;
205             }
206             
207             Class[] intfs = elementType.getInterfaces();
208             for (int i = 0; i < intfs.length; i++) {
209                 componentClassName = registry.get(elementBaseKey + intfs[i].getName());
210                 if (componentClassName != null) {
211                     return componentClassName;
212                 }
213             }
214         }
215         
216         // If we didn't have an elementType or couldn't find a match using elementType, try 
217         // without.
218         return registry.get(baseKey);
219     }

221     /**
222      * Gets the Component for the given type and context. The component is created with a 
223      * BeanPropertyModel, reflecting the given property on the bean. BeanPropertyModel allows
224      * access to the underlying bean.
225      *
226      * @param bean the bean, which may or may not implement IModel.
227      * @param wicketId the Wicket component id.
228      * @param propertyMeta the PropertyMetaData for the property. 
229      * 
230      * @return the viewer Component. 
231      */
232     public Component getComponent(Object bean, String wicketId, ElementMetaData propertyMeta)
233     {
234         boolean viewOnly = propertyMeta.isViewOnly();
235         String componentClassName = propertyMeta.getFieldType();
236         if (componentClassName == null) {
237             Class type = propertyMeta.getPropertyType();
238             // For arrays, treat type as Object[]
239             if (type.isArray()) {
240                 type = Object[].class;
241             }
242             
243             Class elementType = propertyMeta.getElementType(null);
244             
245             // Work up class hierarchy until we find a more generalized component. Also check interface types.
246             for (; type != null && componentClassName == null; type = type.getSuperclass()) {
247                 componentClassName = getComponentClassName(type, elementType);
248                 Class[] intfs = type.getInterfaces();
249                 for (int i = 0; componentClassName == null && i < intfs.length; i++) {
250                     componentClassName = getComponentClassName(intfs[i], elementType);
251                 }
252             }
253         }
254         
255         if (componentClassName != null) {
256             try {
257                 Class componentClass = Class.forName(componentClassName);
258                 Constructor xtor = componentClass.getConstructor(constructorArgs);
259                 
260                 IModel model = new BeanPropertyModel(bean, propertyMeta);
261                 Component component = (Component)xtor.newInstance( new Object[] { wicketId, model, propertyMeta, viewOnly } );
262                 associateLabelToFormComponents(propertyMeta, component);
263                 return component;
264             }
265             catch (ClassNotFoundException e) {
266                 throw new RuntimeException("Component class not found", e);
267             }
268             catch (NoSuchMethodException e) {
269                 throw new RuntimeException("Component class does not implement constructor (String wicketId, IModel model)", e);
270             }
271             catch (Exception e) {
272                 throw new RuntimeException("Error instantiating component " + componentClassName, e);
273             }
274         }
275         
276         // The Object.class registry entry should have caught this, but just in case.
277         return new Label(wicketId, "<No Field for " + propertyMeta.getPropertyType().getName() + ">");
278     }

282     /**
283      * Associate label with FormComponents that are descendants of component. 
284      *
285      * @param propertyMeta
286      * @param component
287      */
288     private void associateLabelToFormComponents(final ElementMetaData propertyMeta, Component component)
289     {
290         if (component instanceof MarkupContainer) {
291             MarkupContainer container = (MarkupContainer)component;
292             container.visitChildren(FormComponent.class, new IVisitor() {
293                 public Object component(Component component) 
294                 {
295                     FormComponent formComponent = (FormComponent)component;
296                     formComponent.setLabel( new Model(propertyMeta.getLabel()) );
297                     formComponent.add( new ErrorHighlightingBehavior() );
298                     return IVisitor.CONTINUE_TRAVERSAL;
299                 }                
300             });
301         }
