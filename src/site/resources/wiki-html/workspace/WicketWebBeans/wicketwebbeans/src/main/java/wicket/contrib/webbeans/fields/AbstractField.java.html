020 package com.googlecode.wicketwebbeans.fields;
021 
022 import wicket.behavior.SimpleAttributeModifier;
023 import com.googlecode.wicketwebbeans.containers.BeanForm;
024 import com.googlecode.wicketwebbeans.model.BeanPropertyModel;
025 import com.googlecode.wicketwebbeans.model.ElementMetaData;
026 import wicket.markup.html.form.FormComponent;
027 import wicket.markup.html.panel.Panel;
028 import wicket.model.IModel;
029 import wicket.util.string.Strings;
030 
031 
032 /**
033  * Base class for Fields. In general, Fields should <em>not</em> set renderBodyOnly to true
034  * if they want to be refreshed via Ajax.
035  *
036  * @author Dan Syrstad
037  */
038 abstract public class AbstractField extends Panel implements Field
039 {
040     private static final long serialVersionUID = -5452855853289381110L;
041     
042     private ElementMetaData elementMetaData;
043 
044     /**
045      * Construct a AbstractField. 
046      *
047      * @param id the Wicket id for the editor.
048      * @param model the model.
049      * @param metaData the meta data for the property.
050      * @param viewOnly true if the component should be view-only.
051      */
052     public AbstractField(String id, IModel model, ElementMetaData metaData, boolean viewOnly)
053     {
054         super(id, model);
055         this.elementMetaData = metaData;
056         
057         metaData.consumeParameter(ElementMetaData.PARAM_REQUIRED);
058         metaData.consumeParameter(ElementMetaData.PARAM_MAX_LENGTH);
059         
060         // Allow for refreshing of the field via Ajax.
061         setOutputMarkupId(true);
062         setRenderBodyOnly(false);
063     }
064     
065     /**
066      * @return true if the field's metadata says that it is a required field.
067      */
068     public boolean isRequiredField()
069     {
070         return elementMetaData.isRequired();
071     }
072     
073     /**
074      * @return the maximum length for the field, or null if no maxmimum length.
075      */
076     public Integer getMaxLength()
077     {
078         return elementMetaData.getMaxLength();
079     }
080     
081     /**
082      * @return the default string value for this field, or null if none is defined.
083      */
084     public String getDefaultValue()
085     {
086         return elementMetaData.getDefaultValue();
087     }
088     
089     /**
090      * Gets a java.text.Format string for formatting the field.
091      *
092      * @return the format, or null if not defined.
093      */
094     public String getFormat()
095     {
096         return elementMetaData.getParameter("format");
097     }
098     
099     protected void setFieldParameters(FormComponent field)
100     {
101         Integer maxLength = getMaxLength();
102         if (maxLength != null) {
103             field.add( new SimpleAttributeModifier("maxlength", maxLength.toString()) );
104         }
105         
106         String defaultValue = getDefaultValue();
107         if (defaultValue != null && Strings.isEmpty(getModelObjectAsString())) {
108             field.setModelValue(defaultValue);
109         }
110     }
111     
112     /**
113      * {@inheritDoc}
114      * @see wicket.Component#onAttach()
115      */
116     @Override
117     protected void onAttach()
118     {
119         super.onAttach();
120         // If we're part of a BeanForm, register ourself with it.
121         BeanForm beanForm  = (BeanForm)findParent(BeanForm.class);
122         if (beanForm != null) {
123             beanForm.registerComponent(this, (BeanPropertyModel)getModel(), elementMetaData);
124         }
125     }
126 
127     /**
128      * Gets the elementMetaData.
129      *
130      * @return a ElementMetaData.
131      */
132     public ElementMetaData getElementMetaData()
133     {
134         return elementMetaData;
135     }
136 
137     /**
138      * Returns the ElementMetaData related to the specified propertyName.
139      * @param metaData metaData for this field.
140      * @param parameterName the parameter name on metaData. The value of this parameter should be the property name to lookup.
141      * @param propertyClass the class type the property should resolve to
142      * @return ElementMetaData
143      */
144     protected ElementMetaData getDependentProperty(ElementMetaData metaData, String parameterName, Class propertyClass) {
145         ElementMetaData property = null;
146         String propStr = metaData.getParameter(parameterName);
147         if (propStr != null) {
148             property = metaData.getBeanMetaData().findElement(propStr);
149             if (property == null) {
150                 throw new RuntimeException("'" + propStr + "' is not defined on " + metaData.getBeanMetaData().getBeanClass());
151             }
152 
153             if (!propertyClass.isAssignableFrom( property.getPropertyType() )) {
154                 throw new RuntimeException(parameterName + "'" + propStr + "' must return a " + propertyClass.getName() + " on "
155                                 + metaData.getBeanMetaData().getBeanClass() + ". Instead it returns "
156                                 + property.getPropertyType());
157             }
158         }
159         return property;
160     }
161 
162     /**
163      * Gets the dependent property bean from the specified property.
164      *
165      * @return the dependent property bean, or null if not defined.
166      * @param property the ElementMetaData property to extract the bean from
167      */
168     protected Object getDependentPropertyBean(ElementMetaData property)
169     {
170         if (property != null) {
171             BeanPropertyModel model = (BeanPropertyModel)getModel();
172             return property.getPropertyValue( model.getBean() );
173         }
174 
175         return null;
176     }
177 }
