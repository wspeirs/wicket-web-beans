020 package com.googlecode.wicketwebbeans.fields;
021 
022 import java.io.Serializable;
023 import java.util.List;
024 
025 import com.googlecode.wicketwebbeans.model.ElementMetaData;
026 import com.googlecode.wicketwebbeans.model.NonJavaEnum;
027 import wicket.markup.ComponentTag;
028 import wicket.markup.MarkupStream;
029 import wicket.markup.html.form.DropDownChoice;
030 import wicket.markup.html.form.IChoiceRenderer;
031 import wicket.markup.html.panel.Fragment;
032 import wicket.model.IModel;
033 import wicket.model.Model;
034 import wicket.util.string.Strings;
035 
036 /**
037  * A field for enumerated types. Presents the values as a drop-down list.
038  * Accepts a parameter of "default" indicating the default choice if the current selection is null.
039  * The default value must match one of the enum.name() values if a Java enum or NonJavaEnum is used. If a {@link IChoiceRenderer}
040  * is used, the default value must match one of those returned by {@link IChoiceRenderer#getIdValue(Object, int)}.
041  * 
042  * @author Dan Syrstad
043  * @author Mark Southern (mrsouthern)
044  */
045 abstract public class EnumField extends AbstractField
046 {
047     private DropDownChoice choice;
048     private IChoiceRenderer choiceRenderer;
049     
050     /**
051      * Construct a new EnumField.
052      *
053      * @param id the Wicket id for the editor.
054      * @param model the model.
055      * @param metaData the meta data for the property.
056      * @param viewOnly true if the component should be view-only.
057      * @param values a List of values to be selected from. The element's toString() is used to 
058      *  produce the value displayed to the user.
059      */
060     public EnumField(String id, IModel model, ElementMetaData metaData, boolean viewOnly, List<?> values)
061     {
062         this(id, model, metaData, viewOnly, new Model((Serializable)values));
063     }
064     
065     /**
066      * Construct a new EnumField.
067      *
068      * @param id the Wicket id for the editor.
069      * @param model the model.
070      * @param metaData the meta data for the property.
071      * @param viewOnly true if the component should be view-only.
072      * @param valueModel an IModel that returns a List of values to be selected from. The element's toString() is used to 
073      *  produce the value displayed to the user.
074      */
075     public EnumField(String id, IModel model, ElementMetaData metaData, boolean viewOnly, IModel valueModel)
076     {
077         this(id, model, metaData, viewOnly, valueModel, null);
078     }
079     
080     /**
081      * Construct a new EnumField.
082      *
083      * @param id the Wicket id for the editor.
084      * @param model the model.
085      * @param metaData the meta data for the property.
086      * @param viewOnly true if the component should be view-only.
087      * @param valueModel an IModel that returns a List of values to be selected from. The element's toString() is used to 
088      *  produce the value displayed to the user.
089      * @param choiceRenderer a renderer that produces the value displayed to the user. May be null to use the default rendering.
090      */
091     public EnumField(String id, IModel model, ElementMetaData metaData, boolean viewOnly, IModel valueModel, final IChoiceRenderer choiceRenderer)
092     {
093         super(id, model, metaData, viewOnly);
094         
095         this.choiceRenderer = choiceRenderer;
096         Fragment fragment;
097         if (viewOnly) {
098             fragment = new Fragment("frag", "viewer");
099             fragment.add(new LabelWithMinSize("component", model) {
100                 @Override
101                 protected void onComponentTagBody(MarkupStream markupStream, ComponentTag openTag)
102                 {
103                     if (choiceRenderer == null) {
104                         super.onComponentTagBody(markupStream, openTag);
105                     }
106                     else {
107                         String value = choiceRenderer.getDisplayValue(getModelObject()).toString();
108                         if (Strings.isEmpty(value)) {
109                             value = "&nbsp;";
110                             setEscapeModelStrings(false);
111                         }
112                         replaceComponentTagBody(markupStream, openTag, value);
113                     }
114                 }
115             });
116         }
117         else {
118             fragment = new Fragment("frag", "editor");
119             choice = new DropDownChoice("component", model, valueModel, choiceRenderer);
120             // Always allow the null choice.
121             choice.setNullValid(true);
122             fragment.add(choice);
123 
124             initDefault();
125         }
126 
127         add(fragment);
128     }
129     
130     /**
131      * Sets the list of possible values. 
132      *
133      * @param values the values.
134      */
135     public void setValues(List<?> values)
136     {
137         setValuesModel( new Model((Serializable)values) );
138     }
139     
140     /**
141      * Sets the model of the list of possible values. This model should return a List. 
142      *
143      * @param valuesModel the values model.
144      */
145     public void setValuesModel(IModel valuesModel)
146     {
147         choice.setChoices(valuesModel);
148         initDefault();
149     }
150     
151     private void initDefault()
152     {
153         // Handle default selection if no model is provided
154         if (choice.getModelObject() == null) {
155             String defaultChoice = getElementMetaData().getParameter("default");
156             if (defaultChoice != null)
157                 setupDefault(defaultChoice);
158         }
159     }
160     
161     private void setupDefault(String defaultChoice)
162     {
163         List<?> values = choice.getChoices();
164         if (values == null || values.isEmpty()) {
165             return;
166         }
167 
168         Object firstValue = values.get(0);
169 
170         // Figure out what type of enum were dealing with
171         boolean isJavaEnum = false;
172         boolean isNonJavaEnum = false;
173         boolean useRenderer = false;
174         if (firstValue instanceof Enum) {
175             isJavaEnum = true;
176         }
177         else if (firstValue instanceof NonJavaEnum) {
178             isNonJavaEnum = true;
179         }
180         else if (choiceRenderer != null) {
181             useRenderer = true;
182         }
183         else {
184             throw new RuntimeException("Unexpected choice type. Must be a Java language Enum, a custom enumeration that implements NonJavaEnum, or must supply an IChoiceRenderer");
185         }
186 
187         // Iterate over the values and find the one that matches the default choice
188         for (int i = 0; i < values.size(); i++) {
189             Object value = values.get(i);
190             String valueStr = "";
191             if (isJavaEnum) {
192                 valueStr = ((Enum<?>)value).name();
193             }
194             else if (isNonJavaEnum) {
195                 valueStr = ((NonJavaEnum)value).name();
196             }
197             else if (useRenderer) {
198                 valueStr = choiceRenderer.getIdValue(value, i);
199             }
200             
201             if (defaultChoice.equals(valueStr)) {
202                 choice.setModelObject(value);
203                 break;
204             }
205         }
206     }
207 }
