261     /**
262      * Rather than using Wicket's required field validation, which doesn't play well with Ajax and forms,
263      * allow validation of fields on actions. User must call this from the action method.
264      * Adds errors to the page if empty required fields are found. 
265      *
266      * @return true if validation was successful, else false if errors were found.
267      */
268     public boolean validateRequired()
269     {
270         RequiredFieldValidator validator = new RequiredFieldValidator();
271         
272         // If we have a tabbed panel, we have to go thru each tab and validate it because the components for a tab are only created
273         // when the tab is open.
274         if (tabbedPanel == null) {
275             visitChildren(AbstractField.class, validator);
276         }
277         else {
278             for (ITab tab : (List<ITab>)tabbedPanel.getTabs()) {
279                 Panel panel = tab.getPanel("x");
280                 // Needs to be part of the page for errors.
281                 getPage().add(panel);
282                 // Cause ListViews to be populated.
283                 panel.internalAttach();
284                 panel.visitChildren(AbstractField.class, validator);
285                 getPage().remove(panel);
286             }
287         }
288         
289         return !validator.errorsFound;
290     }

378     /**
379      * Refresh the targetComponent, in addition to any components that need to be updated
380      * due to property change events.
381      *
382      * @param target
383      * @param targetComponent the targetComponent, may be null.
384      */
385     public void refreshComponents(final AjaxRequestTarget target, Component targetComponent)
386     {
387         if (targetComponent != null) {
388             refreshComponent(target, targetComponent);
389         }
390         
391         if (!refreshComponents.isEmpty()) {
392             // Refresh components fired from our PropertyChangeListener.
393             
394             // Visit all children and see if they match the fired events. 
395             form.visitChildren( new IVisitor() {
396                 public Object component(Component component)
397                 {
398                     Object model = component.getModel();
399                     if (model instanceof BeanPropertyModel) {
400                         BeanPropertyModel propModel = (BeanPropertyModel)model;
401                         ElementMetaData componentMetaData = propModel.getElementMetaData();
402                         for (ComponentPropertyMapping mapping : refreshComponents) {
403                             if (mapping.elementMetaData == componentMetaData) {
404                                 refreshComponent(target, component);
405                                 break;
406                             }
407                         }
408                     }
409 
410                     return IVisitor.CONTINUE_TRAVERSAL;
411                 }
412             });
413 
414             refreshComponents.clear();
415         }
416     }

418     private void refreshComponent(final AjaxRequestTarget target, Component targetComponent)
419     {
420         // Refresh this field. We have to find the parent Field to do this.
421         MarkupContainer field;
422         if (targetComponent instanceof Field) {
423             field = (MarkupContainer)targetComponent;
424         }
425         else {
426             field = targetComponent.findParent(AbstractField.class);
427         }
428         
429         if (field != null) {
430             if (!field.getRenderBodyOnly()) {
431                 target.addComponent(field);
432             }
433             else {
434                 // Field is RenderBodyOnly, have to add children individually
435                 field.visitChildren( new IVisitor() {
436                     public Object component(Component component)
437                     {
438                         if (!component.getRenderBodyOnly()) {
439                             target.addComponent(component);
440                         }
441                         
442                         return IVisitor.CONTINUE_TRAVERSAL;
443                     }
444                     
445                 });
446             }
447         }
448         else {
449             target.addComponent(targetComponent);
450         }

480     private final class FormVisitor implements IVisitor, Serializable
481     {
482         public Object component(Component component) 
483         {
484             if (component instanceof FormComponent) {
485                 boolean addBehavior = true;
486                 for (IBehavior behavior : (List<IBehavior>)component.getBehaviors()) {
487                     if (behavior instanceof FormSubmitter) {
488                         addBehavior = false;
489                         break;
490                     }
491                 }
492                 
493                 if (addBehavior) {
494                     FormSubmitter behavior = new FormSubmitter("onchange");
495                     // Note: Do NOT set a delay. The delay can cause an onchange to be sent AFTER a button submit
496                     // which causes the submit button's messages to be erased. <- That was true when we used AjaxSubmitButtons, we don't anymore.
497                     //behavior.setThrottleDelay(Duration.milliseconds(250));
498                     component.add(behavior);
499                     component.add( new SimpleAttributeModifier("onfocus", "bfOnFocus(this)") );
500                 }
501             }
502             
503             return IVisitor.CONTINUE_TRAVERSAL;
504         }

565      * Simple data structure for mapping components and properties. <p>
566      */
567     private static final class ComponentPropertyMapping implements Serializable
568     {
569         /** IModel holding the bean. */
570         private BeanPropertyModel beanModel;
571         private ElementMetaData elementMetaData;
572         
573         ComponentPropertyMapping(BeanPropertyModel beanModel, ElementMetaData elementMetaData)
574         {
575             this.beanModel = beanModel;
576             this.elementMetaData = elementMetaData;
577         }
578 
579         /** 
580          * {@inheritDoc}
581          * @see java.lang.Object#hashCode()
582          */
583         @Override
584         public int hashCode()
585         {
586             int result = 31 + ((beanModel == null) ? 0 : beanModel.hashCode());
587             result = 31 * result + ((elementMetaData == null) ? 0 : elementMetaData.hashCode());
588             return result;
589         }
590         
591         private Object getBean()
592         {
593             return beanModel.getBean();
594         }
595 
596         /** 
597          * {@inheritDoc}
598          * @see java.lang.Object#equals(java.lang.Object)
599          */
600         @Override
601         public boolean equals(Object obj)
602         {
603             if (!(obj instanceof ComponentPropertyMapping)) {
604                 return false;
605             }
606 
607             final ComponentPropertyMapping other = (ComponentPropertyMapping)obj;
608             return beanModel == other.beanModel && 
609                     (elementMetaData == other.elementMetaData || 
610                      (elementMetaData != null && elementMetaData.equals(other.elementMetaData)));
611         }
612     }

614     /**
615      * Listens to property change events on a bean and adds them to the queue of
616      * components to be refreshed. <p>
617      */
618     public final class BeanPropertyChangeListener implements PropertyChangeListener, Serializable
619     {
620         public void propertyChange(PropertyChangeEvent evt)
621         {
622             // Find matching component
623             Object bean = evt.getSource();
624             String propName = evt.getPropertyName();
625             for (ComponentPropertyMapping mapping : componentPropertyMappings) {
626                 if (bean == mapping.getBean() && propName.equals(mapping.elementMetaData.getPropertyName())) {
627                     BeanForm.this.refreshComponents.add(mapping);
628                 }
629             }
630         }
631     }
