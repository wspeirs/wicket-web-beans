01 package com.googlecode.wicketwebbeans.examples.tables;

03 import java.beans.PropertyChangeListener;
04 import java.beans.PropertyChangeSupport;
05 import java.io.Serializable;
06 import java.math.BigDecimal;
07 import java.math.RoundingMode;
08 
09 public class InvoiceLine implements Serializable 
10 {
11     private static final long serialVersionUID = -6032061214458712439L;
12     
13     public enum ItemCode { Gears, Sprockets, Chains, Wheels, Gizmos };
14 
15     private Integer quantity;
16     private ItemCode itemCode;
17     private BigDecimal cost;
18 
19     private PropertyChangeSupport listeners = new PropertyChangeSupport(this);
20     
21     public InvoiceLine()
22     {
23     }
24 
25     // JavaBeans compliant method to add a PropertyChangeListener. 
26     public void addPropertyChangeListener(PropertyChangeListener listener)
27     {
28         listeners.addPropertyChangeListener(listener);
29     }
30     
31     // JavaBeans compliant method to remove a PropertyChangeListener. 
32     public void removePropertyChangeListener(PropertyChangeListener listener)
33     {
34         listeners.removePropertyChangeListener(listener);
35     }
36 
37     public Integer getQuantity()
38     {
39         return quantity;
40     }
41 
42     public void setQuantity(Integer quantity)
43     {
44         this.quantity = quantity;
45         fireTotalChange();
46     }
47 
48     public ItemCode getItemCode()
49     {
50         return itemCode;
51     }
52 
53     public void setItemCode(ItemCode itemCode)
54     {
55         this.itemCode = itemCode;
56     }
57 
58     public BigDecimal getCost()
59     {
60         return cost;
61     }
62 
63     public void setCost(BigDecimal cost)
64     {
65         // Ensure two decimal places
66         this.cost = (cost == null ? null : cost.setScale(2, RoundingMode.HALF_UP));
67         fireTotalChange();
68     }
69     
70     private void fireTotalChange()
71     {
72         listeners.firePropertyChange("total", null, getTotal());
73     }
74     
75     public BigDecimal getTotal()
76     {
77         return cost == null ? null : cost.multiply( BigDecimal.valueOf((long)quantity) );
78     }
79 }
