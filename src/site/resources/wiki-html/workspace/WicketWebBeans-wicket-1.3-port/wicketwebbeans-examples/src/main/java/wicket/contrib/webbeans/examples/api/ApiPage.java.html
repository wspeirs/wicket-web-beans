01 package com.googlecode.wicketwebbeans.examples.api;
02 
03 import static com.googlecode.wicketwebbeans.annotations.Property.EMPTY;
04 
05 import org.apache.wicket.ajax.AjaxRequestTarget;
06 import org.apache.wicket.markup.html.WebPage;
07 import org.apache.wicket.markup.html.form.Form;
08 
09 import com.googlecode.wicketwebbeans.containers.BeanForm;
10 import com.googlecode.wicketwebbeans.model.BeanMetaData;
11 import com.googlecode.wicketwebbeans.model.api.JAction;
12 import com.googlecode.wicketwebbeans.model.api.JBean;
13 import com.googlecode.wicketwebbeans.model.api.JProperty;
14 import com.googlecode.wicketwebbeans.model.api.JTab;
15 
16 public class ApiPage extends WebPage
17 {
18     public ApiPage()
19     {
20         TestBean bean = new TestBean();
21         
22         // Create the meta data
23         JBean jbean = new JBean(TestBean.class)
24             .tabs(
25                 new JTab("General").propertyNames("firstName", "lastName", "idNumber"),
26                 new JTab("Address")
27                     .propertyNames("address1", EMPTY, EMPTY, 
28                                    "address2", EMPTY, EMPTY, "city", "state", "zip")
29             )
30             // Customize certain properties from above.
31             .properties(
32                 new JProperty("firstName").required(true).maxLength(10),
33                 new JProperty("lastName").required(true)
34             )
35             .actions( new JAction("save").confirm("Are you sure you want to save?") );
36         
37         BeanMetaData meta = new BeanMetaData(bean.getClass(), null, jbean, this, null, false);
38         add( new BeanForm("beanForm", bean, meta) );
39     }
40 
41     public void save(AjaxRequestTarget target, Form form, TestBean bean)
42     {
43         
44         if (!BeanForm.findBeanFormParent(form).validateRequired()) {
45             return; // Errors
46         }
47         
48         info("Saved");
49     }
50 }
