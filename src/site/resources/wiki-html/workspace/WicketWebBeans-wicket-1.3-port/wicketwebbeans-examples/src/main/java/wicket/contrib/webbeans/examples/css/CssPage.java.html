14 public class CssPage extends WebPage
15 {

17     {
18         TestBean bean = new TestBean();
19         
20         // Create the meta data
21         JBeans jbeans = new JBeans(
22             new JBean(TestBean.class)
23                 .columns(1)
24                 .css("greenBeanBorder")
25                 .propertyNames("firstName", "lastName", "rows")
26                 .properties( new JProperty("firstName").css("purpleColor") ),
27             new JBean(RowBean.class)

30                 .properties( new JProperty("description").dynamicCss("getDescriptionCss") )
31         );
32         
33         BeanMetaData meta = new BeanMetaData(bean.getClass(), null, jbeans, this, null, false);
34         add( new BeanForm("beanForm", bean, meta) );
35     }
36 
37     public void save(AjaxRequestTarget target, Form form, TestBean bean)
38     {

54

62             return "fieldNegative";

64
