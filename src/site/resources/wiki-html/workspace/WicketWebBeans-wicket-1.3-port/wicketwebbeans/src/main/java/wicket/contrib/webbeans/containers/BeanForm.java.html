223     /**
224      * Creates the panel for the given tab.
225      *
226      * @param panelId the Wicket id for the panel component.
227      * @param bean may be a bean or an IModel containing a bean.
228      * @param beanMetaData the BeanMetaData.
229      * @param tabMetaData the TabMetaData.
230      * @param container the container class to use. May be null.
231      * 
232      * @return a Panel.
233      */
234     protected Panel createPanel(String panelId, Object bean, BeanMetaData beanMetaData, TabMetaData tabMetaData, Class<? extends Panel> containerClass)
235     {
236         if (containerClass == null) {
237             containerClass = beanMetaData.getContainerClass();
238         }
239         
240         if (containerClass != null) {
241             try {
242                 Constructor<? extends Panel> constructor = containerClass.getConstructor(CONTAINER_CONSTRUCTOR_PARAMS);
243                 return constructor.newInstance(panelId, bean, beanMetaData, tabMetaData);
244             }
245             catch (Exception e) {
246                 throw new RuntimeException("Error instantiating container", e);
247             }
248         }
249         
250         boolean isList = (bean instanceof List);
251         if (bean instanceof IModel) {
252             Object modelBean = ((IModel)bean).getObject();
253             isList = (modelBean instanceof List);
254         }
255         
256         if (isList) {
257             // BeanTablePanel expects a model. Wrap bean if necessary.
258             IModel model;
259             if (bean instanceof IModel) {
260                 model = (IModel)bean;
261             }
262             else {
263                 model = new Model((Serializable)bean);
264             }
265             
266             // Get Number of rows from parameters
267             int rows = beanMetaData.getIntParameter(PARAM_ROWS, 10);
268             return new BeanTablePanel(panelId, model, beanMetaData, rows);
269         }
270 
271         return new BeanGridPanel(panelId, bean, beanMetaData, tabMetaData);
272     }

313     /**
314      * Rather than using Wicket's required field validation, which doesn't play well with Ajax and forms,
315      * allow validation of fields on actions. User must call this from the action method.
316      * Adds errors to the page if empty required fields are found. 
317      *
318      * @return true if validation was successful, else false if errors were found.
319      */
320     public boolean validateRequired()
321     {
322         RequiredFieldValidator validator = new RequiredFieldValidator();
323         
324         // If we have a tabbed panel, we have to go thru each tab and validate it because the components for a tab are only created
325         // when the tab is open.
326         if (tabbedPanel == null) {
327             visitChildren(AbstractField.class, validator);
328         }
329         else {
330             for (ITab tab : (List<ITab>)tabbedPanel.getTabs()) {
331                 Panel panel = tab.getPanel("x");
332                 // Needs to be part of the page for errors.
333                 getPage().add(panel);
334                 // Cause ListViews to be populated.
335                 panel.beforeRender();
336                 panel.visitChildren(AbstractField.class, validator);
337                 getPage().remove(panel);
338             }
339         }
340         
341         return !validator.errorsFound;
342     }

430     /**
431      * Refresh the targetComponent, in addition to any components that need to be updated
432      * due to property change events.
433      *
434      * @param target
435      * @param targetComponent the targetComponent, may be null.
436      */
437     public void refreshComponents(final AjaxRequestTarget target, Component targetComponent)
438     {
439         if (targetComponent != null) {
440             refreshComponent(target, targetComponent);
441         }
442         
443         if (!refreshComponents.isEmpty()) {
444             // Refresh components fired from our PropertyChangeListener.
445             
446             // Visit all children and see if they match the fired events. 
447             form.visitChildren( new IVisitor() {
448                 public Object component(Component component)
449                 {
450                     Object model = component.getModel();
451                     if (model instanceof BeanPropertyModel) {
452                         BeanPropertyModel propModel = (BeanPropertyModel)model;
453                         ElementMetaData componentMetaData = propModel.getElementMetaData();
454                         for (ComponentPropertyMapping mapping : refreshComponents) {
455                             if (mapping.elementMetaData == componentMetaData) {
456                                 refreshComponent(target, component);
457                                 break;
458                             }
459                         }
460                     }
461 
462                     return IVisitor.CONTINUE_TRAVERSAL;
463                 }
464             });
465 
466             refreshComponents.clear();
467         }
468     }

470     private void refreshComponent(final AjaxRequestTarget target, Component targetComponent)
471     {
472         // Refresh this field. We have to find the parent Field to do this.
473         MarkupContainer field;
474         if (targetComponent instanceof Field) {
475             field = (MarkupContainer)targetComponent;
476         }
477         else {
478             field = targetComponent.findParent(AbstractField.class);
479         }
480         
481         if (field != null) {
482             if (!field.getRenderBodyOnly()) {
483                 target.addComponent(field);
484             }
485             else {
486                 // Field is RenderBodyOnly, have to add children individually
487                 field.visitChildren( new IVisitor() {
488                     public Object component(Component component)
489                     {
490                         if (!component.getRenderBodyOnly()) {
491                             target.addComponent(component);
492                         }
493                         
494                         return IVisitor.CONTINUE_TRAVERSAL;
495                     }
496                     
497                 });
498             }
499         }
500         else {
501             target.addComponent(targetComponent);
502         }
503     }

651          */
652         @Override

660             return beanModel == other.beanModel &&

662                      (elementMetaData != null && elementMetaData.equals(other.elementMetaData)));
663         }

682         }
