020 package com.googlecode.wicketwebbeans.fields;
021 
022 import java.io.Serializable;
023 import java.util.List;
024 
025 import org.apache.wicket.markup.ComponentTag;
026 import org.apache.wicket.markup.MarkupStream;
027 import org.apache.wicket.markup.html.form.DropDownChoice;
028 import org.apache.wicket.markup.html.form.IChoiceRenderer;
029 import org.apache.wicket.markup.html.panel.Fragment;
030 import org.apache.wicket.model.IModel;
031 import org.apache.wicket.model.Model;
032 import org.apache.wicket.util.string.Strings;
033 
034 import com.googlecode.wicketwebbeans.model.ElementMetaData;
035 import com.googlecode.wicketwebbeans.model.NonJavaEnum;
036 
037 /**
038  * A field for enumerated types. Presents the values as a drop-down list.
039  * Accepts a parameter of "default" indicating the default choice if the current selection is null.
040  * 
041  * @author Dan Syrstad
042  */
043 abstract public class EnumField extends AbstractField
044 {
045     private DropDownChoice choice;
046     private IChoiceRenderer choiceRenderer;
047     
048     /**
049      * Construct a new EnumField.
050      *
051      * @param id the Wicket id for the editor.
052      * @param model the model.
053      * @param metaData the meta data for the property.
054      * @param viewOnly true if the component should be view-only.
055      * @param values a List of values to be selected from. The element's toString() is used to 
056      *  produce the value displayed to the user.
057      */
058     public EnumField(String id, IModel model, ElementMetaData metaData, boolean viewOnly, List<?> values)
059     {
060         this(id, model, metaData, viewOnly, new Model((Serializable)values));
061     }
062     
063     /**
064      * Construct a new EnumField.
065      *
066      * @param id the Wicket id for the editor.
067      * @param model the model.
068      * @param metaData the meta data for the property.
069      * @param viewOnly true if the component should be view-only.
070      * @param valueModel an IModel that returns a List of values to be selected from. The element's toString() is used to 
071      *  produce the value displayed to the user.
072      */
073     public EnumField(String id, IModel model, ElementMetaData metaData, boolean viewOnly, IModel valueModel)
074     {
075         this(id, model, metaData, viewOnly, valueModel, null);
076     }
077     
078     /**
079      * Construct a new EnumField.
080      *
081      * @param id the Wicket id for the editor.
082      * @param model the model.
083      * @param metaData the meta data for the property.
084      * @param viewOnly true if the component should be view-only.
085      * @param valueModel an IModel that returns a List of values to be selected from. The element's toString() is used to 
086      *  produce the value displayed to the user.
087      * @param choiceRenderer a renderer that produces the value displayed to the user. May be null to use the default rendering.
088      */
089     public EnumField(String id, IModel model, ElementMetaData metaData, boolean viewOnly, IModel valueModel, final IChoiceRenderer choiceRenderer)
090     {
091         super(id, model, metaData, viewOnly);
092         
093         this.choiceRenderer = choiceRenderer;
094         Fragment fragment;
095         if (viewOnly) {
096             fragment = new Fragment("frag", "viewer");
097             fragment.add(new LabelWithMinSize("component", model) {
098                 @Override
099                 protected void onComponentTagBody(MarkupStream markupStream, ComponentTag openTag)
100                 {
101                     if (choiceRenderer == null) {
102                         super.onComponentTagBody(markupStream, openTag);
103                     }
104                     else {
105                         String value = choiceRenderer.getDisplayValue(getModelObject()).toString();
106                         if (Strings.isEmpty(value)) {
107                             value = "&nbsp;";
108                             setEscapeModelStrings(false);
109                         }
110                         replaceComponentTagBody(markupStream, openTag, value);
111                     }
112                 }
113             });
114         }
115         else {
116             fragment = new Fragment("frag", "editor");
117             choice = new DropDownChoice("component", model, valueModel, choiceRenderer);
118             // Always allow the null choice.
119             choice.setNullValid(true);
120             fragment.add(choice);
121 
122             initDefault();
123         }
124 
125         add(fragment);
126     }
127     
128     /**
129      * Sets the list of possible values. 
130      *
131      * @param values the values.
132      */
133     public void setValues(List<?> values)
134     {
135         setValuesModel( new Model((Serializable)values) );
136     }
137     
138     /**
139      * Sets the model of the list of possible values. This model should return a List. 
140      *
141      * @param valuesModel the values model.
142      */
143     public void setValuesModel(IModel valuesModel)
144     {
145         choice.setChoices(valuesModel);
146         initDefault();
147     }
148     
149     private void initDefault()
150     {
151         // Handle default selection if no model is provided
152         if (choice.getModelObject() == null) {
153             String defaultChoice = getElementMetaData().getParameter("default");
154             if (defaultChoice != null)
155                 setupDefault(defaultChoice);
156         }
157     }
158     
159     private void setupDefault(String defaultChoice)
160     {
161         List<?> values = choice.getChoices();
162         if (values == null || values.isEmpty()) {
163             return;
164         }
165 
166         Object firstValue = values.get(0);
167 
168         // Figure out what type of enum were dealing with
169         boolean isJavaEnum = false;
170         boolean isNonJavaEnum = false;
171         boolean useRenderer = false;
172         if (firstValue instanceof Enum) {
173             isJavaEnum = true;
174         }
175         else if (firstValue instanceof NonJavaEnum) {
176             isNonJavaEnum = true;
177         }
178         else if (choiceRenderer != null) {
179             useRenderer = true;
180         }
181         else {
182             throw new RuntimeException("Unexpected choice type. Must be a Java language Enum, a custom enumeration that implements NonJavaEnum, or must supply an IChoiceRenderer");
183         }
184 
185         // Iterate over the values and find the one that matches the default choice
186         for (int i = 0; i < values.size(); i++) {
187             Object value = values.get(i);
188             String valueStr = "";
189             if (isJavaEnum) {
190                 valueStr = ((Enum<?>)value).name();
191             }
192             else if (isNonJavaEnum) {
193                 valueStr = ((NonJavaEnum)value).name();
194             }
195             else if (useRenderer) {
196                 valueStr = choiceRenderer.getIdValue(value, i);
197             }
198             
199             if (defaultChoice.equals(valueStr)) {
200                 choice.setModelObject(value);
201                 break;
202             }
203         }
204     }
