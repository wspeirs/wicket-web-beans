17 package com.googlecode.wicketwebbeans.model;
18 
19 import com.googlecode.wicketwebbeans.model.NonJavaEnum;
20 
21 import java.io.Serializable;
22 import java.util.List;
23 
24 /**
25  * Base implementation of NonJavaEnum to make it look similar to a Java Enum.
26  * 
27  * @author Marc Stock
28  * @author Dan Syrstad
29  */
30 public abstract class BaseNonJavaEnum implements NonJavaEnum, Serializable
31 {
32     protected String name;
33     protected String displayValue;
34 
35     public BaseNonJavaEnum(String name, String displayValue)
36     {
37         this.name = name;
38         this.displayValue = displayValue;
39     }
40 
41     public String name()
42     {
43         return name;
44     }
45 
46     public String getDisplayValue()
47     {
48         return displayValue;
49     }
50 
51     public void setDisplayValue(String displayValue)
52     {
53         this.displayValue = displayValue;
54     }
55 
56     /**
57      * Get the Enum for the given name.
58      *
59      * @param enumValue name to match
60      *
61      * @return a Enum, or null if not found.
62      * @param enums cachedEnums to search through
63      */
64     public static BaseNonJavaEnum valueOf(String enumValue, List<? extends BaseNonJavaEnum> enums)
65     {
66         if (enumValue == null)
67             return null;
68 
69         for (BaseNonJavaEnum nonJavaEnum : enums) {
70             if (nonJavaEnum.name().equals(enumValue)) {
71                 return nonJavaEnum;
72             }
73         }
74 
75         return null;
76     }
77 
78     @Override
79     public String toString()
80     {
81         return getDisplayValue();
82     }
83 
84     @Override
85     public boolean equals(Object obj)
86     {
87         if (this == obj) {
88             return true;
89         }
90 
91         if (!(obj instanceof BaseNonJavaEnum)) {
92             return false;
93         }
94 
95         BaseNonJavaEnum other = (BaseNonJavaEnum)obj;
96         return name().equals(other.name());
97     }
98 }
