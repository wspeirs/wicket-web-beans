20 package com.googlecode.wicketwebbeans.fields;
21 
22 import com.googlecode.wicketwebbeans.containers.BeanGridPanel;
23 import com.googlecode.wicketwebbeans.model.BeanMetaData;
24 import com.googlecode.wicketwebbeans.model.ElementMetaData;
25 import org.apache.wicket.model.IModel;
26 
27 
28 /**
29  * A Field that presents a property's bean inline with other fields.
30  *
31  * @author Dan Syrstad
32  */
33 public class BeanInlineField extends AbstractField implements UnlabeledField
34 {
35     /**
36      * Construct a new BeanGridField.
37      *
38      * @param id the Wicket id for the editor.
39      * @param model the model.
40      * @param metaData the meta data for the property.
41      * @param viewOnly true if the component should be view-only.
42      */
43     public BeanInlineField(String id, IModel model, ElementMetaData metaData, boolean viewOnly)
44     {
45         super(id, model, metaData, viewOnly);
46         
47         BeanMetaData beanMetaData = metaData.createBeanMetaData(viewOnly);
48         
49         if (!viewOnly && model.getObject() == null) {
50             // Create a blank instance for editing.
51             model.setObject( metaData.createInstance() );
52         }
53 
54         add( new BeanGridPanel("beanPanel", model, beanMetaData) );
55     }
56 }
